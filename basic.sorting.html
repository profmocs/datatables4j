Basic initialisation

Table sorting

By default, all columns are always sortable.

You can enable (or disable) the sort feature using the sort table attribute.

Example 1
<datatables:table id="myTableId" data="${persons}" sort="false">
    <datatables:column title="Id" property="id" />
    <datatables:column title="FirstName" property="firstName" />
    <datatables:column title="LastName" property="lastName" />
    <datatables:column title="Street" property="address.street1" />
    <datatables:column title="Mail" property="mail" />
 </datatables:table>

As a result, DataTables4j will produce and externalize the following Javascript code :

var oTable_myTableId;
$(document).ready(function(){
    oTable_myTableId = $('#myTableId').dataTable({
        "sDom" : "lfrtip",
        "bSort" : false,
        "aoColumns" : [
            {"bSortable" : true}, 
            {"bSortable" : true}, 
            {"bSortable" : true}, 
            {"bSortable" : true}, 
            {"bSortable" : true} 
        ]
    });
});
Example 2

Moreover, you can use the sortable column attribute to enable sorting on a particular column.

<datatables:table id="myOtherTableId" data="${persons}">
    <datatables:column title="Id" property="id" sortable="false"/>
    <datatables:column title="FirstName" property="firstName" />
    <datatables:column title="LastName" property="lastName" />
    <datatables:column title="Street" property="address.street1" />
    <datatables:column title="Mail" property="mail" sortable="false"/>
 </datatables:table>

As a result, DataTables4j will produce and externalize the following Javascript code :

var oTable_myOtherTableId;
$(document).ready(function(){
    oTable_myOtherTableId = $('#myOtherTableId').dataTable({
        "sDom" : "lfrtip",
        "aoColumns" : [ 
            {"bSortable" : false},
            {"bSortable" : true}, 
            {"bSortable" : true}, 
            {"bSortable" : true}, 
            {"bSortable" : false}
        ]
    });
});
